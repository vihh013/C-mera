function capturarFoto() { var options = { quality: 50, destinationType: Camera.DestinationType.FILE_URI, sourceType: Camera.PictureSourceType.CAMERA, encodingType: Camera.EncodingType.JPEG, saveToPhotoAlbum: true }; navigator.camera.getPicture(onFotoCapturada, onErroCapturaFoto, options); } } function onDeviceReady() { console.log("Dispositivo pronto."); } function onFotoCapturada(imageURI) { console.log("Imagem capturada: " + imageURI); } function onErroCapturaFoto(error) { console.error("Erro ao capturar a imagem: " + error); }
